dijkstra.cpp 
void dijkstra(int start) {
  priority_queue<pair<int,int>> pq;
  pq.push({0, start});
  while (!pq.empty()) {
    auto [d, u] = pq.top();
    if (d > dist[u]) continue;
    for (auto [v, w] : adj[u])
      if (dist[v] > dist[u] + w) {
        dist[v] = dist[u] + w;
        pq.push({dist[v], v});
      }
  }
}
